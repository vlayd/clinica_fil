No ListModel:
    Tabs na labela:
     #[Override]
    public function getTabs(): array
    {
        return [
            'todos' => Tab::make(label: 'Todos')
                ->icon(Heroicon::ListBullet)
                ->badge(User::count())
                ->modifyQueryUsing(fn(Builder $query) => $query),
            'ativos' => Tab::make(label: 'Ativos')
                ->icon(Heroicon::CheckCircle)
                ->badge(User::where('active', 1)->count())
                ->badgeColor('success')
                ->modifyQueryUsing(fn(Builder $query) => $query->where('active', true)),
            'inativos' => Tab::make(label: 'Inativos')
                ->badge(User::where('active', 0)->count())
                ->badgeColor('danger')
                ->icon(Heroicon::XCircle)
                ->modifyQueryUsing(fn(Builder $query) => $query->where('active', false))
                ->deferBadge(); //Para não atualizar ao clicar, só atualizar com refresh da página
        ];
    }
