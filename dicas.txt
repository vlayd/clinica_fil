Em AdminPanelProvider, em $panel
->font('nome da fonte') //Altera a fonte
->passwordReset() //Acrescenta no reset senha no login
->revealablePasswords(false) //Tira o olhinho do password
->sidebarCollapsibleOnDesktop() //Para contrair a sidebar
->sidebarFullyCollapsibleOnDesktop() //Se transforma de menu sanduiche
->topNavigation() //Os menus ficam em cima e não do lado
->profile() //Vai aparecer o opção do profile para editar o perfil logado
->brandLogo(asset('images/person_login.png')) //Logo
->brandName('Clínicas') //Nome da sidebar
->brandLogoHeight('3rem')
->colors() //Só pedir o google para criar a sua
->favicon()
->sidebarWidth('15rem') //Largura da sidebar (esquerda)
->maxContentWidth(Width::Full) //Largura do conteúdo (direita)
->spa() //Single Page Aplication. Os menus abrem rápido, sem reload

    protected static ?string $navigationLabel = 'Usuários'; // Alterar o texto na sidebar
    protected static ?int $navigationSort = 1; // A ordem na sidebar, no caso será o 1, dashbard sempre é o primeiro
